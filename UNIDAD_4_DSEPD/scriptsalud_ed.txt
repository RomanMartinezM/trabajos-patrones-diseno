CREATE DATABASE Nutricion;

CREATE TABLE USUARIO(
IDPERSONA INTEGER PRIMARY KEY NOT NULL
GENERATED ALWAYS AS IDENTITY
(START WITH 1 INCREMENT BY 1),
NOMBRE VARCHAR(30),
FECHANAC Date,
ESTATURA INT,
SEXO char,
PASSWORD varchar(30)
);

CREATE TABLE ACTIVIDADUSUARIO
(
IDATVUSER INTEGER PRIMARY KEY NOT NULL
GENERATED ALWAYS AS IDENTITY
(START WITH 1 INCREMENT BY 1),
DESCRIPCION VARCHAR(250),
FECHA1 Date,
FECHA2 Date,
USUARIO INT REFERENCES USUARIO(IDPERSONA),
ACTIVIDAD INT REFERENCES TIPOACTIVIDAD(IDACTIVIDAD)
);
create table TIPOACTIVIDAD
(
IDACTIVIDAD INTEGER PRIMARY KEY NOT NULL
GENERATED ALWAYS AS IDENTITY
(START WITH 1 INCREMENT BY 1),
DESCRIPCION VARCHAR(50)NOT NULL,
FACTORH FLOAT NOT NULL,
FACTORM FLOAT NOT NULL,
USUARIO INT REFERENCES USUARIO(IDPERSONA)

);
create table antro
(
idantro serial primary key not null,
usuario integer references usuario(idpersona),
fecha date,
peso int,
cintura int,
cadera int
);
